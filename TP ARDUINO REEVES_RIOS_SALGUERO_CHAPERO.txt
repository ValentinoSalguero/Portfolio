#include <LiquidCrystal.h>
//Declaramos variales 
int vector[4]={0,0,0,0};
int botonInicio=10;
int botonFin=7;
int buzzer=8;
int inicio=0;
int fin=0;
float tiempo=0;
long T1=0;
long T2=0;
float T3=0;
int i=0;

LiquidCrystal lcd(12, 11, 5, 4, 3, 2);

void setup() {
  Serial.begin(9600);
  lcd.begin(16, 2);
  pinMode(botonInicio, INPUT_PULLUP);
  pinMode(buzzer, OUTPUT);
  pinMode(botonFin, INPUT_PULLUP);
}
void loop() {
  
  inicio=digitalRead(botonInicio);
  
  if (inicio==HIGH){
    tiempo=cuantoTiempo();
    if(tiempo>0.05 && tiempo<0.3){vector[i]=1;
                  i=i+1;}
    if(tiempo>0.3){vector[i]=2;
                   i=i+1;}
  }
  
  fin=digitalRead(botonFin);
  
  if(fin==HIGH){
    EscribirLetra();
    i=0;
    vector[0]=0; vector[1]=0; vector[2]=0; vector[3]=0;
  }
} //FIN LOOP

float cuantoTiempo() {
  inicio=digitalRead(botonInicio);
  if(inicio==HIGH){
    T1=millis(); //T inicial
    while(inicio==HIGH){
      digitalWrite(buzzer,HIGH);
      T2=millis(); //T Final
      inicio=digitalRead(botonInicio);
      }
    digitalWrite(buzzer,LOW);
    T3=T2-T1; //Delta Tiempo
    T3=T3/1000; //CONVERSION A SEGUNDOS 
    Serial.println(T3);
  	}
  return T3;
  T3=0;
  }
